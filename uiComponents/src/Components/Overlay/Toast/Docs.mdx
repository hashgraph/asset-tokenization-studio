## How to use

Default use:

```tsx dark
import { useToast } from "@iob/io-bricks-ui/Components/Toast";

const Component = () => {
  const toast = useToast();

  const handleClick = () => {
    toast.show({
      title: "Title",
      description: "Description",
      status: "success",
    });
  };

  return <Button onClick={handleClick}>Show Toast</Button>;
};
```

Show custom toast:

```tsx dark
import { useToast } from "@iob/io-bricks-ui/Components/Toast";

const Component = () => {
  const toast = useToast();

  const handleClick = () => {
    toast.show({
      title: "Title",
      description: "Description",
      status: "success",
      render: (props) => (
        <CustomToast
          {...props}
          title="Custom Title"
          description="Custom Description"
          status="success"
        />
      ),
    });
  };

  return <Button onClick={handleClick}>Show Toast</Button>;
};

const CustomToast = ({ title, description, status, onClose }) => {
  return (
    <Box
      bg={status === "success" ? "green.500" : "red.500"}
      color="neutral"
      p={3}
      borderRadius="md"
      pos="relative"
    >
      <Text fontWeight="bold">{title}</Text>
      <Text>{description}</Text>
      <Button onClick={onClose}>Close</Button>
    </Box>
  );
};
```

Render global toast component:

```tsx dark
import {
  useToast,
  ToastConfigurator,
} from "@iob/io-bricks-ui/Components/Toast";

const Component = () => {
  const toast = useToast();

  const handleClick = () => {
    toast.show({
      title: "Title",
      description: "Description",
      status: "success",
    });
  };

  return (
    <>
      <ToastConfigurator component={ToastComponent} />
      <Button onClick={handleClick}>Show Toast</Button>
    </>
  );
};
```
